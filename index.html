function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var params = JSON.parse(e.postData.contents);
  
  // Agrega una fila nueva mapeando exactamente los 19 ítems de gestión de Coaching
  sheet.appendRow([
    new Date(params.fecha), 
    params.entrenador, 
    params.turno,
    params.p1 ? "SÍ" : "NO",
    params.p2 ? "SÍ" : "NO",
    params.p3 ? "SÍ" : "NO",
    params.p4 ? "SÍ" : "NO",
    params.p5 ? "SÍ" : "NO",
    params.p6 ? "SÍ" : "NO",
    params.p7 ? "SÍ" : "NO",
    params.p8 ? "SÍ" : "NO",
    params.p9 ? "SÍ" : "NO",
    params.p10 ? "SÍ" : "NO",
    params.p11 ? "SÍ" : "NO",
    params.p12 ? "SÍ" : "NO",
    params.p13 ? "SÍ" : "NO",
    params.p14 ? "SÍ" : "NO",
    params.p15 ? "SÍ" : "NO",
    params.p16 ? "SÍ" : "NO",
    params.p17 ? "SÍ" : "NO",
    params.p18 ? "SÍ" : "NO",
    params.p19 ? "SÍ" : "NO",
    params.score + "%", 
    params.observacionesIA
  ]);
  
  return ContentService.createTextOutput(JSON.stringify({"status": "success", "row": sheet.getLastRow()})).setMimeType(ContentService.MimeType.JSON);
}
3
